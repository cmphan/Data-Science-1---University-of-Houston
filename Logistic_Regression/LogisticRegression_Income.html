#Calculate confusion matrix precision

def precision_recall(y_test, y_pred):
    cm = confusion_matrix(y_test, y_pred)    
    tp = cm[0,0]
    fp = cm[0,1]
    fn = cm[1,0]
    prec = tp / (tp+fp)
    rec = tp / (tp+fn)    
    return prec, rec
precision, recall = precision_recall(y_test, y_pred)
print('Precision: %f Recall %f' % (precision, recall))
